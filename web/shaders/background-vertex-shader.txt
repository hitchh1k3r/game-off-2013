precision mediump float;

attribute vec2 aVertexPosition;

uniform mat4 uVMatrix;
uniform int uRender3D;

varying vec4 vSkyPoint;

void main()
{
	gl_Position =  vec4(aVertexPosition, 0.0, 1.0);
	vSkyPoint = uVMatrix * vec4(normalize(vec3(aVertexPosition.x, (uRender3D == 1 ? 1.0 : -1.0) * aVertexPosition.y, -1.0)), 0.00005);
}
